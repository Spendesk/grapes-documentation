import { AnimationTokens } from "./animation-tokens";

export const renderAnimationsToken = (animationToken) => {
  const value = getComputedStyle(document.documentElement).getPropertyValue(
    animationToken
  );
  return (
    <div key={animationToken} className="text-center">
      <div className={styles.title}>{animationToken}</div>
      <div className={styles.body}>{value}</div>
    </div>
  );
};

## Usage in CSS

Whenever you want to apply this style, you can use the corresponding variable.

```css
.myClass {
  animation-duration: var(--transition-duration);
  animation-timing-function: var(--transition-timing-function);
}
```
