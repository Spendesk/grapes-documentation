const [selectedValue, setSelectedValue] = useState<string | null>(null);
return (
  <OptionGroup
    options={[
      { value: "daily", label: "Daily" },
      { value: "weekly", label: "Weekly" },
      { value: "yearly", label: "Yearly" },
    ]}
    value={selectedValue}
    name="optionGroupField"
    onChange={(event) => {
      setSelectedValue(event.target.value);
    }}
  />
);
