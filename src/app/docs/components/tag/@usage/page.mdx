import { Avatar, Tag } from "@dev-spendesk/grapes";
import { Preview } from "@/components/preview/preview";

## Variant

The Tag comes with six visual styles: alert, info (default), neutral, primary, success, and warning.

<Preview>
  <Tag variant="alert">Alert</Tag>
  <Tag variant="info">Info</Tag>
  <Tag variant="neutral">Neutral</Tag>
  <Tag variant="primary">Primary</Tag>
  <Tag variant="success">Success</Tag>
  <Tag variant="warning">Warning</Tag>
</Preview>
```tsx
<Tag variant="alert">Alert tag</Tag>
<Tag variant="info">Info tag</Tag>
<Tag variant="neutral">Neutral tag</Tag>
<Tag variant="primary">Primary tag</Tag>
<Tag variant="success">Success tag</Tag>
<Tag variant="warning">Warning tag</Tag>
```

## Use an icon

Use the `iconName` prop to set a icon on the left side of the Tag.

<Preview>
  <Tag variant="primary" iconName="airplane">
    My tag
  </Tag>
</Preview>

```tsx
<Tag variant="primary" iconName="airplane">
  My tag
</Tag>
```

## Rounded tag

Use the `isRounded` prop to round the tag.

<Preview>
  <Tag variant="neutral" isRounded>
    <Avatar variant="circle" src="/aurelien.webp" size="xs" text="aurelien" />
    <span className="ml-xs">Alex Smith</span>
  </Tag>
</Preview>

```tsx
<Tag variant="neutral" isRounded>
  <Avatar variant="circle" src="/aurelien.webp" size="xs" text="aurelien" />
  <span className="ml-xs">Alex Smith</span>
</Tag>
```
