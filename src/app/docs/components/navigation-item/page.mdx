import { Badge, NavigationItem } from "@dev-spendesk/grapes";
import { Preview } from "@/lib/preview/preview";

:::info{title="Usage"}
The NavigationItem component should only be used in the [Navigation component](/docs/components/navigation).
:::

## Addon

The `leftAddon` and `rightAddon` props are used to display another element inside the navigation item.

<Preview>
  <NavigationItem text="Navigation item" leftAddon={<Badge>12</Badge>} />
  <NavigationItem text="Navigation item" rightAddon={<Badge>12</Badge>} />
</Preview>

```tsx sandbox
<>
  <NavigationItem leftAddon={<Badge>12</Badge>} text="Navigation item" />
  <NavigationItem rightAddon={<Badge>12</Badge>} text="Navigation item" />
</>
```

## Active

The `isActive` prop is used to display the navigation item as active.

<Preview className="items-center">
  <NavigationItem isActive text="Main navigation item" />
</Preview>

```tsx sandbox
<>
  <NavigationItem isActive text="Main navigation item" />
</>
```
