import { Preview } from "@/components/preview/preview";

## Basic usage

<Preview>
  <TextArea className="w-[360px]" placeholder="Describe your expense" />
</Preview>

```tsx
const [value, setValue] = useState("");

return (
  <TextArea
    placeholder="Describe your expense"
    value={value}
    onChange={(event) => {
      setValue(event.target.value);
    }}
  />
);
```

## Rows

The `rows` prop allows you to specify how many rows the textarea should have. Its default value is 3.

<Preview>
  <TextArea
    className="w-[360px]"
    placeholder="Describe your expense"
    rows={10}
  />
</Preview>

```tsx
<TextArea rows={10} />
```

## Accessibility

Adheres to the [textbox role requirements](https://www.w3.org/TR/wai-aria-1.2/#textbox).
